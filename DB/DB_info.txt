(unsigned)
	tiny  : 0 to 255
	small : 0 to 65535

product(상품 정보)
	name(품명) varchar(20)
	code(코드) tinyint unsigned primary
	price(가격) smallint unsigned
	expiration(유통기한) smallint unsigned



currency(권액 정보)
	code(코드) tinyint unsigned primary
	currency(권액) smallint unsigned



stock_001(001자판기 재고)
	p_code(코드) tinyint unsigned foreign key(product.code)
	stock(재고량) tinyint unsigned
	mf_date(제조일자) date
	ep_date(유통기한) date




stock_002(002자판기 재고)
	p_code(코드) tinyint unsigned foreign key(product.code)
	stock(재고량) tinyint unsigned
	mf_date(제조일자) date
	ep_date(유통기한) date




sales(판매량)
	p_code(코드) tinyint unsigned foreign key(product.code)
	sold(판매일) date




bill(권액별 재고)
	c_code(권액 코드) tinyint unsigned
	stock(매수) tinyint unsigned
